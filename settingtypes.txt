# Options:
# - default: Store rankings in modstorage, not recommended for servers
# - redis: Store rankings in a redis DB, recommended for servers, requires some setup
ctf_rankings_backend (Rankings Backend) enum default default,redis

# Port the redis server is hosted on.
# This setting is only used when ctf_rankings_backend = redis
ctf_rankings_redis_server_port (Rankings Redis Server Port) int 6379

# CTF can sometimes reduce CPU usage by calculating things once and putting them in RAM.
# Turn on low ram mode if you're getting crashes related to RAM/Memory usage
ctf_low_ram_mode (Low RAM mode) bool false

# Enables the CTF shop formspec.
enable_ctf_shop (Enable CTF Shop) bool true
ctf_feedback_path (Path to save CTF feedbacks) string
ctf_feedback_min_playtime (Minimum playtime required to give feedback) number

#The interval in which the server_status.json file is updated.
#Default: 20 seconds. Lower values increase the server load.
server_status.update_interval (Server status update interval in seconds) float 20

#Entire path to the output file. If empty, the default value
#<worldpath>/server_status.json will be used.
#Example: /home/luanti/server_status.json
server_status.filepath (Server status save path for JSON-file) string

#The interval in which the list of "alive" players in the rush mode.
#Default: 5 seconds. Lower values increase the server load.
spectator_teleport.update_interval List of alive players update interval in seconds) float 5
